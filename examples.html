Sass - Color function


// Service functions
@each $theme, $color in $ui-colors {
    .service--#{$theme} {

        .service__inner {
            .icon-container {
                background-color: $color;
                border-color: $color;
            }
        }

        &:hover {
            .service__inner {
                background-color: $color;
                color: $color-white;

                .icon-container {
                    span {
                        color: $color !important;
                    }
                }

                [class^="btn--"],
                button {
                    color: $color ;
                }
            }
        }
    }
}
